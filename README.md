# Khởi tạo kho lưu trữ git mới:
# git init
# Đặt giá trị cấu hình cho tên người dùng và email của bạn:
# git config --global user.name <tên của bạn>
# git config --global user.email <email của bạn>
# Sao chép một kho lưu trữ:
# git clone <repository-url>
# Thêm một tập tin vào khu vực tổ chức:
# git add <tập tin>
# Thêm tất cả các thay đổi tệp vào khu vực tổ chức:
# git add .
# Kiểm tra các thay đổi chưa được thực hiện:
# git diff
# Cam kết các thay đổi theo giai đoạn:
# git commit -m "Tin nhắn"
# Đặt lại khu vực tổ chức về lần xác nhận cuối cùng:
# git reset
# Kiểm tra trạng thái của thư mục làm việc và khu vực tổ chức:
# git status
# Xóa một tệp khỏi chỉ mục và thư mục làm việc:
# git rm <tập tin>
# Liệt kê lịch sử cam kết:
# git log
# Kiểm tra các thay đổi siêu dữ liệu và nội dung của cam kết:
# git show <commit-hash>
# Liệt kê tất cả các chi nhánh địa phương:
# git branch
# Tạo một chi nhánh mới:
# git branch <tên nhánh>
# Đổi tên chi nhánh hiện tại:
# git branch -m <tên nhánh mới>
# Xóa một nhánh:
# git branch -d <tên nhánh>
# Chuyển sang chi nhánh khác:
# git check out <tên chi nhánh>
# Hợp nhất nhánh được chỉ định vào nhánh hiện tại:
# git merge <tên nhánh>
# Tạo kết nối mới tới kho lưu trữ từ xa:
# git remote add <tên> <repository-url>
# Đẩy các thay đổi đã cam kết vào kho lưu trữ từ xa:
# git push <điều khiển> <nhánh>
# Tải xuống nội dung từ kho lưu trữ từ xa:
# git pull <điều khiển từ xa>
# Dọn dẹp các tệp không cần thiết và tối ưu hóa kho lưu trữ cục bộ:
# git gc
# Tạm thời loại bỏ những thay đổi không được cam kết và
# git stash
# Áp dụng lại các thay đổi đã lưu trữ trước đó
# git stash apply
